public class Main {

    public static void main(String[] args)
    {
    	Scanner in = new Scanner(System.in);

        System.out.println("Enter the days::\n");

        int d = in.nextInt();
        int y;
        int w;

        // d = days
        // y = years
        // w = weeks

        /* Conversion of days in to years, weeks and days */
        y = (d / 365);
        w = (d % 365) / 7;
        d = d - ((y * 365) + (w));

    }
